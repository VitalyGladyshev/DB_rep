2. При помощи базы данных Redis решите задачу поиска имени пользователя по электронному
адресу и наоборот, поиск электронного адреса пользователя по его имени.

127.0.0.1:6379> SET Alex$alex@mail.ru 0
OK
127.0.0.1:6379> SET Boris$boris@mail.ru 1
OK
127.0.0.1:6379> SET Victor$victor@mail.ru 2
OK
127.0.0.1:6379> KEYS Boris*
1) "Boris$boris@mail.ru"
127.0.0.1:6379> KEYS *victor@mail.ru
1) "Victor$victor@mail.ru"
127.0.0.1:6379>
